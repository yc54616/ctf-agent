    34: ===== src/buildroot/build_script.sh =====
    35: #!/bin/bash
    36: 
    37: BUILDROOT=$(pwd)
    38: 
    39: # Build kernel & rootfs
    40: make
    41: 
    42: # Copy kernel image into /output
    43: mkdir -p /output
    44: cp $BUILDROOT/output/images/Image.gz /output/Image.gz
    45: 
    46: # Trim rootfs
    47: mkdir /rootfs
    48: cd /rootfs
    49: gunzip -c $BUILDROOT/output/images/rootfs.cpio.gz | cpio -vid
    50: 
    51: # Copy init script & mdev.conf into rootfs
    52: cp /init init && chmod 755 init
    53: cp /mdev.conf etc/mdev.conf && chmod 644 etc/mdev.conf
    54: 
    55: # Create rootfs with fake flag (public)
    56: echo 'DH{fake_flag}' > flag
    57: chmod 400 flag
    58: find . | cpio -H newc --owner root -o | gzip > /output/rootfs.cpio.gz
    59: 
    60: cd /output
    61: rm -rf /rootfs
    62: 
    63: pwd && ls -alF
    64: echo 'Done!'
