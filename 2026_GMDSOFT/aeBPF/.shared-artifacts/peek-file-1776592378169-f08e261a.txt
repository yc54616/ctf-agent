    54: ===== build_script.sh =====
    55: #!/bin/bash
    56: 
    57: BUILDROOT=$(pwd)
    58: 
    59: # Build kernel & rootfs
    60: make
    61: 
    62: # Copy kernel image into /output
    63: mkdir -p /output
    64: cp $BUILDROOT/output/images/Image.gz /output/Image.gz
    65: 
    66: # Trim rootfs
    67: mkdir /rootfs
    68: cd /rootfs
    69: gunzip -c $BUILDROOT/output/images/rootfs.cpio.gz | cpio -vid
    70: 
    71: # Copy init script & mdev.conf into rootfs
    72: cp /init init && chmod 755 init
    73: cp /mdev.conf etc/mdev.conf && chmod 644 etc/mdev.conf
    74: 
    75: # Create rootfs with fake flag (public)
    76: echo 'DH{fake_flag}' > flag
    77: chmod 400 flag
    78: find . | cpio -H newc --owner root -o | gzip > /output/rootfs.cpio.gz
    79: 
    80: cd /output
    81: rm -rf /rootfs
    82: 
    83: pwd && ls -alF
    84: echo 'Done!'
